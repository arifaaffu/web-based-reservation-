// Define the beer menu items (sample data) const beerMenuItems = [ { name: "Craft Brew 1", type: "Craft Brew", price: "₹200" }, { name: "Craft Brew 2", type: "Craft Brew", price: "₹220" }, // Add more items as needed ]; // Define the food menu items (sample data) const foodMenuItems = [ { name: "Burger 1", category: "Burger", price: "₹150" }, { name: "Burger 2", category: "Burger", price: "₹180" }, // Add more items as needed ]; // Function to generate beer menu HTML function generateBeerMenu() { const beerMenuSection = document.getElementById("beer-menu"); beerMenuItems.forEach(item => { beerMenuSection.innerHTML += `
${item.name} - ${item.type} - ${item.price}
`; }); } // Function to generate food menu HTML function generateFoodMenu() { const foodMenuSection = document.getElementById("food-menu"); foodMenuItems.forEach(item => { foodMenuSection.innerHTML += `
${item.name} - ${item.category} - ${item.price}
`; }); } // Call functions to generate menus when the page loads window.onload = function() { generateBeerMenu(); generateFoodMenu(); };
